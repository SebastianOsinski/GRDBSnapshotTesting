CREATE TRIGGER author_update
    AFTER UPDATE ON author
    WHEN NEW.lastUpdate = OLD.lastUpdate
    BEGIN
        UPDATE place SET lastUpdate = CURRENT_TIMESTAMP WHERE id = NEW.id;
    END
